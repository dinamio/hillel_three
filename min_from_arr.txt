public static int min(int[] arr) {
                if (arr.length > 0){
                        int min = arr[0];
                        for (int i = 0; i < arr.length; i++) {
                                if (min > arr[i]) {
                                        min = arr[i];
                                }
                        }
                        return min;
                }
                else throw new IllegalArgumentException();

        }
